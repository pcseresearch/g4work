174 void SBSDetectorConstruction::SetMagField(G4double fieldValue)
175 {
176  //apply a global uniform magnetic field along Z axis
177  G4FieldManager* fieldMgr
178  = G4TransportationManager::GetTransportationManager()->GetFieldManager();
179 
180  if(magField) delete magField; //delete the existing magn field
181 
182  if(fieldValue!=0.) // create a new one if non nul
183  { magField = new G4UniformMagField(G4ThreeVector(0.,0.,fieldValue));
184  fieldMgr->SetDetectorField(magField);
185  fieldMgr->CreateChordFinder(magField);
186  } else {
187  magField = 0;
188  fieldMgr->SetDetectorField(magField);
189  }
190 }
